class aide::syslog (
  Stdlib::Absolutepath $logdir = $::aide::logdir
) {
  assert_private()
  include '::rsyslog'

  rsyslog::rule::local { 'XX_aide':
    rule            => '$programname == \'aide\'',
    target_log_file => "${logdir}/aide.log",
    stop_processing => true
  }
}
