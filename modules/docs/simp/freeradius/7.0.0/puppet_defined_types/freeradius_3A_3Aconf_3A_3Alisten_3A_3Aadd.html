  $listen_type,
  $ipaddr             = '*',
  $port               = '0',
  $interface          = 'nil',
  $per_socket_clients = ''
) {

  file { "/etc/raddb/conf/listen.inc/${name}":
    ensure  => 'file',
    owner   => 'root',
    group   => 'radiusd',
    content => template('freeradius/conf/listen.erb'),
    notify  => Service['radiusd']
  }
