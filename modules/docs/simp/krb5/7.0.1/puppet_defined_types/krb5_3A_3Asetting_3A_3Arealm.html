This define allows you to add a realm to the [realms] section of /etc/krb5.conf

Parameters:

  • name (String) —

    The affected Realm. This will be upcased.

  • admin_server (Simplib::Host) —

    The host where the admin server is running.

  • kdc (Optional[Simplib::Host]) (defaults to: undef) —

    The host where the KDC is running.

  • default_domain (Optional[String]) (defaults to: undef) —

    The default domain in which hosts are assumed to be present.

  • v4_instance_convert (Hash[String,String]) (defaults to: {}) —

    A hash of ‘tag name’ to ‘tag value’ mappings for default domain mapping translations.

  • v4_realm (Optional[String]) (defaults to: undef) —

    The v4 realm to be used when talking to legacy systems.

  • auth_to_local_names (Hash[String,String]) (defaults to: {}) —

    A hash of ‘principal names’ to ‘local user names’ per the man page.

  • auth_to_local (Optional[String]) (defaults to: undef) —

    A general rule for mapping to local user names. The following values are allowed: DB: RULE: DEFAULT

  • target (Stdlib::Absolutepath) (defaults to: pick(getvar('::krb5::config::config_dir'), '/etc/krb5.conf.d'))

define krb5::setting::realm (
  Simplib::Host                 $admin_server,
  Optional[Simplib::Host]       $kdc                 = undef,
  Optional[String]              $default_domain      = undef,
  Hash[String,String]           $v4_instance_convert = {},
  Optional[String]              $v4_realm            = undef,
  Hash[String,String]           $auth_to_local_names = {},
  Optional[String]              $auth_to_local       = undef,
  Stdlib::Absolutepath          $target              = pick(getvar('::krb5::config::config_dir'), '/etc/krb5.conf.d'),
) {

  if !defined(Class['krb5']) {
    fail('You must include ::krb5 before using ::krb5::setting::realm')
  }

  $_name = munge_krb5_conf_filename($name)

  file { "${target}/${_name}__realm":
    owner   => 'root',
    group   => 'root',
    mode    => '0644',
    seltype => 'krb5_conf_t',
    content => template("${module_name}/realm.erb")
  }

  Class['krb5'] -> Krb5::Setting::Realm[$name]
}
