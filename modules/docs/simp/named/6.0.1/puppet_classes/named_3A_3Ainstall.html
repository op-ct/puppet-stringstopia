class named::install (
  Boolean              $chroot      = true,
  Stdlib::Absolutepath $chroot_path = $::named::chroot_path,
  String               $ensure      = 'latest'
){
  assert_private()

  group { 'named':
    ensure    => 'present',
    allowdupe => false,
    gid       => '25'
  }

  user { 'named':
    ensure     => 'present',
    allowdupe  => false,
    uid        => '25',
    gid        => '25',
    home       => '/var/named',
    membership => 'inclusive',
    shell      => '/sbin/nologin',
    require    => Group['named']
  }

  if $::osfamily == 'RedHat' {
    package { 'bind': ensure => $ensure }
    package { 'bind-libs': ensure => $ensure }

    if $chroot {
      package { 'bind-chroot': ensure => $ensure }
    }
    else {
      package { 'bind-chroot': ensure => 'absent' }
    }
  }
  else {
    fail("Operating System ${::operatingsystem} is not yet supported")
  }
}
