class pam::wheel (
  String           $wheel_group   = 'wheel',
  Boolean          $root_only     = false,
  Boolean          $use_openshift = $::pam::use_openshift,
  Optional[String] $content       = $::pam::su_content
) inherits ::pam {
  if $content {
    $_content = $content
  }
  else {
    $_content = template("${module_name}/etc/pam.d/su.erb")
  }

  file { '/etc/pam.d/su':
    owner   => 'root',
    group   => 'root',
    mode    => '0644',
    content => $_content
  }
}
