define pupmod::master::autosign (
  String $entry
) {

  $l_name = inline_template("<%= '${name}'.gsub('/','_') %>")

  simpcat_fragment { "autosign+${l_name}.autosign":
    content => "${name}\n${entry}\n"
  }
}
