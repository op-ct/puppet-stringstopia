class simp::mountpoints::proc (
  Integer[0,2]      $proc_hidepid = 2,
  Optional[Integer] $proc_gid     = undef
) {
  if $proc_gid {
    $_proc_options = "hidepid=${proc_hidepid},gid=${proc_gid}"
  }
  else {
    $_proc_options = "hidepid=${proc_hidepid}"
  }

  mount { '/proc':
    ensure   => 'mounted',
    atboot   => true,
    device   => 'proc',
    fstype   => 'proc',
    remounts => true,
    options  => $_proc_options
  }
}
