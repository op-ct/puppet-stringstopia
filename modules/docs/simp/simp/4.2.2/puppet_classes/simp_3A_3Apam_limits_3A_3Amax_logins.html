class simp::pam_limits::max_logins (
  Integer[0] $value = 10
) {

  simplib::assert_metadata( $module_name )

  pam::limits::rule { 'max_logins':
    domains => ['*'],
    type    => 'hard',
    item    => 'maxlogins',
    value   => $value,
    order   => 100
  }
}
